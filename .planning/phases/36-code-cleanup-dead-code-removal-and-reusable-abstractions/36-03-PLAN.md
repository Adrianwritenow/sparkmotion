    ```typescript
    /**
     * Throws FORBIDDEN if a CUSTOMER user tries to access a resource outside their org.
     * ADMIN users always pass. Called explicitly in procedures — not middleware (per locked decision).
     */
    export function enforceOrgAccess(
      ctx: { user: { role: string; orgId?: string | null } | null },
      entityOrgId: string
    ): void {
      if (!ctx.user) throw new TRPCError({ code: "UNAUTHORIZED" });
      if (ctx.user.role === "CUSTOMER" && entityOrgId !== ctx.user.orgId) {
        throw new TRPCError({ code: "FORBIDDEN" });
      }
    }
    ```

    **Step 1 — Create createTrashProcedures factory in packages/api/src/lib/trash.ts:**

    This factory generates the 4 trash procedures (trashCount, listDeleted, restore, restoreAll) for any entity. Entity-specific behavior is passed via options callbacks.

    ```typescript
    import { z } from "zod";
    import { TRPCError } from "@trpc/server";
    import { db } from "@sparkmotion/database";
    import { protectedProcedure } from "../trpc";
    import { DELETED } from "./soft-delete";

    interface TrashProcedureOptions {
      model: "event" | "campaign" | "band" | "organization";
      /** Select fields for listDeleted query */
      selectFields: Record<string, boolean | object>;
      /** How to get orgId from the entity (for CUSTOMER access control) */
      getOrgId: (entity: any) => string;
      /** Whether this entity is admin-only (organizations) */
      adminOnly?: boolean;
      /** Custom restore logic (runs inside transaction). Return value becomes mutation result. */
      onRestore?: (id: string, entity: any, ctx: any) => Promise<any>;
      /** Custom restoreAll logic */
      onRestoreAll?: (entities: any[], ctx: any) => Promise<any>;
    }
    ```

    The factory returns an object with 4 procedures:

    **trashCount:** Count soft-deleted records. ADMIN sees all. CUSTOMER sees org-scoped. If adminOnly, use adminProcedure.

    **listDeleted:** List soft-deleted records with deletedBy user name resolution. Same org-scoping as trashCount. Accepts optional orgId input for admin filtering.

    **restore:** Find record, verify not already active, verify org access, then either call onRestore callback or do default restore (set deletedAt=null, deletedBy=null).

    **restoreAll:** Find all deleted records (org-scoped for CUSTOMER), either call onRestoreAll callback or do default batch restore. Return { restored: count }.

    IMPORTANT: The factory must handle these entity-specific differences:

    - **events**: onRestore restores cascade-deleted bands + campaignId FK from deletedCampaignId. onRestoreAll does batch campaign FK restoration.
    - **bands**: onRestore checks compound unique [bandId, eventId] conflict before restoring. Returns { restored, skipped }. Calls invalidateBandCache after.
    - **organizations**: adminOnly=true. onRestore cascade-restores events + campaigns + bands. onRestoreAll cascade-restores all children.
    - **campaigns**: No special logic. Default restore/restoreAll suffice.

    The factory accesses `db[model]` dynamically. Since Prisma client uses `db.event`, `db.campaign`, `db.band`, `db.organization`, use a type-safe accessor pattern:

    ```typescript
    const modelClient = db[options.model as keyof typeof db] as any;
    ```

    Yes, this uses `any` — the type safety comes from the factory's own interface and the tests. This is a pragmatic choice that keeps the factory simple per KISS.

    **Step 2 — Refactor events router to use factory:**

    In events.ts, remove the inline trashCount, listDeleted, restore, restoreAll procedures. Replace with:

    ```typescript
    import { createTrashProcedures } from "../lib/trash";

    export const eventsRouter = router({
      // ... existing procedures (list, byId, create, update, softDelete, etc.) ...

      ...createTrashProcedures({
        model: "event",
        selectFields: {
          id: true, name: true, deletedAt: true, deletedBy: true,
          orgId: true, org: { select: { name: true } },
        },
        getOrgId: (event) => event.orgId,
        onRestore: async (id, event, ctx) => {
          await db.$transaction([
            db.event.update({
              where: { id },
              data: {
                deletedAt: null, deletedBy: null,
                ...(event.deletedCampaignId ? { campaignId: event.deletedCampaignId, deletedCampaignId: null } : {}),
              },
            }),
            db.band.updateMany({
              where: { eventId: id, deletedAt: { gte: event.deletedAt } },
              data: { deletedAt: null, deletedBy: null },
            }),
          ]);
          return { success: true };
        },
        onRestoreAll: async (events, ctx) => {
          await db.$transaction(async (tx) => {
            const eventsWithCampaign = events.filter((e: any) => e.deletedCampaignId);
            for (const e of eventsWithCampaign) {
              await tx.event.update({
                where: { id: e.id },
                data: { campaignId: e.deletedCampaignId, deletedCampaignId: null },
              });
            }
            const eventIds = events.map((e: any) => e.id);
            await tx.event.updateMany({
              where: { id: { in: eventIds } },
              data: { deletedAt: null, deletedBy: null },
            });
            await tx.band.updateMany({
              where: { eventId: { in: eventIds }, deletedAt: { not: null } },
              data: { deletedAt: null, deletedBy: null },
            });
          });
          return { restored: events.length };
        },
      }),
    });
    ```

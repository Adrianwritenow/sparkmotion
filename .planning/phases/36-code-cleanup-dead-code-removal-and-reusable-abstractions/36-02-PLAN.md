    **Step 3 — Refactor all 7 trash buttons to use TrashSheet:**

    Each trash button file shrinks to ~30-40 lines: just tRPC hooks wiring + TrashSheet render.

    Pattern for each entity (using events as example):

    ```typescript
    "use client";
    import { TrashSheet, type TrashItem } from "@sparkmotion/ui";
    import { trpc } from "@/lib/trpc";
    import { toast } from "sonner";

    export function EventTrashButton() {
      const utils = trpc.useUtils();
      const { data: count = 0 } = trpc.events.trashCount.useQuery();
      const [open, setOpen] = useState(false);
      const { data: items = [], isLoading } = trpc.events.listDeleted.useQuery(undefined, { enabled: open });

      const restore = trpc.events.restore.useMutation({
        onSuccess: () => {
          toast.success("Event restored");
          utils.events.trashCount.invalidate();
          utils.events.listDeleted.invalidate();
          utils.events.list.invalidate();
        },
      });

      const restoreAll = trpc.events.restoreAll.useMutation({
        onSuccess: (result) => {
          toast.success(`${result.restored} events restored`);
          // ... invalidations
        },
      });

      const trashItems: TrashItem[] = items.map((e) => ({
        id: e.id,
        displayName: e.name,
        deletedAt: e.deletedAt!,
        deletedByName: e.deletedByName,
        subtitle: e.org?.name, // admin only
      }));

      return (
        <TrashSheet
          label="Events"
          count={count}
          isLoading={isLoading}
          items={trashItems}
          onRestore={(id) => restore.mutate({ id })}
          onRestoreAll={() => restoreAll.mutate()}
          isRestoring={restore.isPending}
          isRestoringAll={restoreAll.isPending}
          showDeletedBy={true}  // admin version
          showRestoreAll={true}
        />
      );
    }
    ```
